<!-- Leaflet JS -->
<script src="https://unpkg.com/leaflet/dist/leaflet.js"></script>

<script>
    // -------------------------------
    // MAP INITIALIZATION
    // -------------------------------
    let map = L.map("map").setView([20.5937, 78.9629], 5); // India center

    L.tileLayer("https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png", {
        maxZoom: 19,
    }).addTo(map);

    // Fetch weather by clicking map
    map.on("click", (e) => {
        fetch("/weather_by_coords", {
            method: "POST",
            headers: { "Content-Type": "application/json" },
            body: JSON.stringify({ lat: e.latlng.lat, lon: e.latlng.lng }),
        })
            .then((res) => res.json())
            .then((data) => displayWeather(data));
    });

    // -------------------------------
    // WEATHER BY CITY
    // -------------------------------
    function searchCity() {
        let city = document.getElementById("city").value;

        fetch("/weather_by_city", {
            method: "POST",
            headers: { "Content-Type": "application/json" },
            body: JSON.stringify({ city: city }),
        })
            .then((res) => res.json())
            .then((data) => displayWeather(data));
    }

    // -------------------------------
    // DISPLAY WEATHER
    // -------------------------------
    function displayWeather(data) {
        let box = document.getElementById("weather-box");

        if (data.cod != 200) {
            box.style.display = "block";
            box.innerHTML = `<b>Error:</b> City not found.`;
            return;
        }

        box.style.display = "block";
        box.innerHTML = `
            <h3>${data.name}, ${data.sys.country}</h3>
            <img src="https://openweathermap.org/img/wn/${data.weather[0].icon}@2x.png">
            <p><b>Temperature:</b> ${data.main.temp}°C</p>
            <p><b>Feels Like:</b> ${data.main.feels_like}°C</p>
            <p><b>Weather:</b> ${data.weather[0].description}</p>
            <p><b>Humidity:</b> ${data.main.humidity}%</p>
            <p><b>Wind Speed:</b> ${data.wind.speed} m/s</p>
        `;
    }

    // -------------------------------
    // AUTOCOMPLETE SUGGESTIONS
    // -------------------------------
    function showSuggestions(text) {
        let box = document.getElementById("suggestions");

        if (text.length < 2) {
            box.innerHTML = "";
            return;
        }

        fetch(`https://geocode.maps.co/search?q=${text}`)
            .then((res) => res.json())
            .then((data) => {
                box.innerHTML = "";
                data.slice(0, 5).forEach((item) => {
                    let div = document.createElement("div");
                    div.innerHTML = item.display_name;
                    div.onclick = () => {
                        document.getElementById("city").value = item.display_name;
                        box.innerHTML = "";
                    };
                    box.appendChild(div);
                });
            });
    }
</script>
